server {
		listen        443 ssl;
		listen        [::]:443 ssl;

        server_name ws.supersecuredomain.com;

		root /var/www/ws.juandmegon.com/public;

        index index.php;

        ssl_certificate /etc/nginx/certs/ws.supersecuredomain.com/fullchain.pem;
        ssl_certificate_key /etc/nginx/certs/ws.supersecuredomain.com/key.pem;

        location / {
                try_files $uri $uri/ /index.php?$query_string;
        }

        location ~ \.php$ {
                include snippets/fastcgi-php.conf;

                fastcgi_pass unix:/var/run/php/php7.4-fpm.sock;
        }

        location /app {
                proxy_pass             http://127.0.0.1:6001;
                proxy_read_timeout     60;
                proxy_connect_timeout  60;
                proxy_redirect         off;

                # Allow the use of websockets
                proxy_http_version 1.1;
                proxy_set_header Upgrade $http_upgrade;
                proxy_set_header Connection 'upgrade';
                proxy_set_header Host $host;
                proxy_cache_bypass $http_upgrade;
        }
}
